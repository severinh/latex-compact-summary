\newcommand{\seeslides}[1][]{%
	\verbose{%
		\colorbox{Cerulean!50}{%
			\textcolor{black}{%
				See slides%
				\ifthenelse{\isempty{#1}}{}{: #1}%
			}%
		}%
	}%
}

\newcommand{\seeexercise}[1][]{%
	\verbose{%

			\textcolor{black}{%
				See exercise%
				\ifthenelse{\isempty{#1}}{}{: #1}%
			}%
		}%
	}%
}

\newcommand{\seeliterature}[1][]{%
	\verbose{%
		\colorbox{Cerulean!50}{%
			\textcolor{black}{%
				See literature%

		}%
	}%
}

\newcommand{\unclear}[1][]{%

	}%
